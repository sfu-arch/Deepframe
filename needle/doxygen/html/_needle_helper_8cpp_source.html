186  // Create a new global variable, 2 words per store (addr+data)

189  // the undo log

197  for (auto &S : Stores) {
198  if (AA.isMustAlias(MemoryLocation::get(SI), MemoryLocation::get(S)))
199  return true;
200  }
201  return false;

206  for (auto &I : **BB) {

208  // Filter out the stores added due to live outs
209  // being returned as a struct by reference.

212  if (!isAliasingStore(SI))

303  ReturnInst::Create(Context, ConstantInt::getFalse(Context), RetFalseBlock);
